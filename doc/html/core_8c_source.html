35/** @brief this is the worker thread routine
36 *
37 * @param arg pointer to data structure holding all relevant data and return data fields
38 * @return returns NULL, returned data is in result field of WorkerData struct
39 */
40int WorkerRoutine( worker_data_t * arg ) {
41 #define NSIM 100
42 #define ELAPSED ((finish.tv_sec - start.tv_sec) + 1.0e-9 * (finish.tv_nsec - start.tv_nsec))
43
44 /* get a copy of the game_info and card_info structs,
45 * as we need to change it during Simulation */
46 GameInfo game_info = *(arg->game_info);
47
48 /* record start time */
49 struct timespec start, finish;
50 clock_gettime(CLOCK_MONOTONIC, &start);
51 finish = start;
52
53 while ( ELAPSED < arg->time_goal && arg->stop == 0 ) {
54 /* get a fresh copy of the game info */
55 game_info = *(arg->game_info);
56
57 /* randomly distribute the cards */
58 mc_sample(&game_info, arg->card_info, &arg->random_seed);
59 arg->mc_sample_calls++;
60
61 /* simulate a game for each legal card and collect the results */
62 for ( uint8_t c = 0; c < arg->legal_cards_len; c++ ) {
63 GameInfo game_info_copy = game_info;
64 arg->results[c] += Simulate(&game_info_copy, arg->legal_cards[c], &arg->random_seed);
65 arg->simulate_calls++;
66 }
67 clock_gettime(CLOCK_MONOTONIC, &finish);
68 }
69
70 arg->time_goal = ELAPSED;
71 return 0;
72}

168/** @brief Execute a move.
169 *
170 * This will handle all internal things that happen when someone plays a card. In case the machine
171 * itself plays a card, this simply removes it from the cardset[0] and updates the trick and scores.
172 * In case of human player this also processes information about the card played, such as for
173 * example if the player does not have a specific suit.
174 *
175 * @param game_info Pointer to GameInfo object to operate on.
176 * @param card_info Pointer to CardInfo object to operate on.
177 * @param card Card Id to play. Will not check if player has card.
178 */
179void ExecuteMove( GameInfo* game_info, CardInfo* card_info, CardId card ) {
180 if ( game_info->next == 0 ) {
181 /* perform machine move */
182 PlayCard( game_info, card );
183 } else {
184 /* perform human move */
185
186 /* remove the played card from the list of cards */
187 for ( uint8_t c = 0; c < card_info->cards_left; c++ ) {
188 if ( card_info->ids[c] == card ) {
189 memmove(&card_info->ids[c], &card_info->ids[c+1], sizeof(card_info->ids[0])*(card_info->cards_left - c - 1));
190 memmove(&card_info->scores[c], &card_info->scores[c+1], sizeof(card_info->scores[0])*(card_info->cards_left - c - 1));
191 card_info->cards_left--;
192 card_info->player_left[game_info->next - 1]--;
193 break;
194 }
195 }
196
197 PlayCard( game_info, card );
198
199 /* fixme dirty fix for that play card subtracts from cardset */
200 for ( uint8_t p = 1; p < 4; p++ ) {
201 game_info->player_cardsets[p] = 0ul;
202 }

204 }
